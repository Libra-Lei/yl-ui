定制组件

# npm
npm i @libra-lei/yl-ui
# OR yarn
yarn add @libra-lei/yl-ui

# 引入

# 完整引入

import Vue from 'vue';

// 完整引入
import YlUI from '@libra-lei/yl-ui'
Vue.use(YlUI)

// 或者按需引入
import { Radio, UploadMini } from '@libra-lei/yl-ui';
Vue.component(Radio.name, Radio);
Vue.component(UploadMini.name, UploadMini);
